	        dr[i] = new DoorController(); 
	    }

		System.out.println(dr[0].isRightDoor());
		System.out.println(dr[1].isRightDoor());
		System.out.println(dr[2].isRightDoor());

		dialogMaster.setVisible(true);
		winCount.setVisible(true);
		rewardTotal.setVisible(true);
		jb[0].setText("一號門");
		jb[1].setText("二號門");
		jb[2].setText("三號門");
	}
	
	public void doorClose() 
	{
		for(int i =0 ; i< dr.length;i++)
		{
			boolean o = dr[i].getRightDoor();
			boolean f = dr[i].isFirstSel();
			if(o == false && f == false)
			{
	            jb[i].setVisible(false); 
	            return;
			}
		}
	}
	
	public void reStartGame()
	{
		firstSelect = false;
		lastChance.setText("剩餘機會: "+ chance );
		dialogMaster.setText("請選擇一扇門");
		winCount.setText("累計勝場:" + winCounter);
		int rewardFinal = rewardGain * winCounter;
		rewardTotal.setText("累計獎金: " + rewardFinal);
		startInGame();
		closeOpenJb(true);
		
		
	}
	
	public void closeOpenJb(boolean b)
	{
		for(int i =0 ; i< jb.length ; i++)
		{
			jb[i].setVisible(b);
		}
	}
	
	public void reSetGame()
	{
		chance = 0;
		winCounter = 0;
		rewardGain = 0;
		lastChance.setText("剩餘機會: "+ chance );
		dialogMaster.setText("請選擇一扇門");
		winCount.setText("累計勝場:" + winCounter);

		dialogMaster.setVisible(false);
		winCount.setVisible(false);
		rewardTotal.setVisible(false);
		gameIsStart = false;
		firstSelect = false;
		startButton.setVisible(true);
		surrenderButton.setVisible(false);
		
		jb[0].setText("一號門");
		jb[1].setText("二號門");
		jb[2].setText("三號門");
		
	}
